using namespace std;

int main(int argc, char const *argv[])
{
    int ii, casenum;
    int h, m;

    scanf("%d", &casenum);
    for (ii = 0; ii < casenum; ii++)
    {
        scanf("%d:%d", &h, &m);
        if (m == 0)
        {
            if (h == 12)
            {
                printf("24\n");
            }
            else
            {
                printf("%d\n", (12 + h) % 24);
            }
        }
        else
        {
            printf("0\n");
        }
    }
    return 0;
}
