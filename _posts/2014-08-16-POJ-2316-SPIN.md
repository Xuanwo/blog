源地址：

http://poj.org/problem?id=2316

# 理解
所有数据相加，再对10取模，因为每个字符串的长度都是相同的～

int main(int argc, char const *argv[])
{
    int a[11] = {0}, i;
    char b[11];
    while (cin >> b)
    {
        for (i = 0; i < strlen(b); i++)
            a[i] += b[i] - '0';
    }
    for (i = 0; i < strlen(b); i++)
    {
        a[i] %= 10;
        cout << a[i];
    }
    cout << endl;
    return 0;
}
