POJ 2840 Big Clock

2014-08-18
Code
Comments

题目

源地址：

理解

很有意思的题目，求解大本钟敲几下的问题，按照规律来。

代码


#include <cstdio>
#include <cstdlib>
#include <cstring>

#include <fstream>
#define debug puts("-----")
#define pi (acos(-1.0))
#define eps (1e-8)
#define inf (1<<28)
using namespace std;

int main(int argc, char const *argv[])
{
    int ii, casenum;
    int h, m;

    scanf("%d", &casenum);
    for (ii = 0; ii < casenum; ii++)
    {
        scanf("%d:%d", &h, &m);
        if (m == 0)
        {
            if (h == 12)
            {
                printf("24\n");
            }
            else
            {
                printf("%d\n", (12 + h) % 24);
            }
        }
        else
        {
            printf("0\n");
        }
    }
    return 0;
}
