#define inf (1<<28)
using namespace std;

#define MAX 300

int main(int argc, char const *argv[])
{
    char input[MAX];
    cin.getline(input, MAX);
    int count = 0;
    char *a , *b;
    a = strtok(input, " ");
    b = strtok(NULL, " ");

    for (int i = 0; i < strlen(a); i++)
    {
        for (int j = 0; j < strlen(b); j++)
        {
            count += (a[i] - '0') * (b[j] - '0');
        }
    }
    printf("%d\n", count);
    return 0;
}
