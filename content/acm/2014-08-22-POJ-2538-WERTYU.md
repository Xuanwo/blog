int main(int argc, char const *argv[])
{
    char ch;

    init();
    while ((ch = getchar()) != EOF)
    {
        if (ch != ' ' && ch != '\n')

            putchar(str[ch]);
        }
        else
        {
            putchar(ch);
        }
    }
    return 0;
}
