int main(int argc, char const *argv[])
{
    int i, n; double x, y, c, sum;
    while (cin >> n)
    {
        sum = 0;
        cin >> x >> y;
        for (i = 1; i <= n; i++)
        {
            cin >> c;
            sum += 2 * (n + 1 - i) * c;
        }
        sum = (n * x + y - sum) / (n + 1);
        printf("%.2f\n", sum);
    }
    return 0;
}
