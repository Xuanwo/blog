# 理解
水题。
一个简单的递归，实际上有一个小小的Trick，因为题目中说明了到目前为止没有证明，所以返回值为-1的情况完全可以不用考虑。

using namespace std;

int t,n;

int func(int n)
{
    int step=0;
    while(n!=1)
    {
        if(n%2==0)  n=n/2;
        else    n=3*n+1;
        step++;
    }
    return step;
}

int main()
{
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d", &n);
        printf("%d\n", func(n));
    }

}
